

```sh
# 3. Extract server info (version, protocol, capabilities)
nmap -p 3306 192.168.177.129 --script=mysql-info
```

```sh
# 4. List all user accounts (needs creds: mysqluser/mysqlpass)
nmap -p 3306 192.168.177.129 --script=mysql-users --script-args="mysqluser='root' , mysqlpass=''"
```

```sh
# 5. List all databases accessible to user
nmap -p 3306 192.168.177.129 --script=mysql-databases --script-args="mysqluser='root' , mysqlpass=''"
```

```sh
# 6. Extract config variables (max_connections, buffers, paths)
nmap -p 3306 192.168.177.129 --script=mysql-variables --script-args="mysqluser='root' , mysqlpass=''"
```
